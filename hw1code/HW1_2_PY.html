import random
import math
import numpy as np
import matplotlib.pyplot as plt

c = 1.36879;
g = [];
j = 0;
nn = 100;
for i in range(1,1000):
    x = random.expovariate(0.5);
    u = random.random();
    if 4*(1/math.sqrt(math.pi))*math.sqrt(x)*math.exp(-0.5*x) / c >= u:
        g.append(x);
        i = i + 1;
        j = j + 1;
    
import scipy.special as sps

shape, scale = 1.5, 1.  # mean=4, std=2*sqrt(2)
count, bins, ignored = plt.hist(g, nn, normed=True)
y = bins**(shape-1)*(np.exp(-bins/scale) / (sps.gamma(shape)*scale**shape))
plt.plot(bins, y, linewidth=2, color='r')
plt.title('Desired pdf in red / Generated pmf in blue', fontsize=18)
plt.show()
